Exemplo de Código SQL Formatado (Clientes)

Boas práticas incluem indentação e comentários.


-- Definição da Tabela de Clientes

CREATE TABLE Clientes (
    -- Identificador único para cada cliente
    ClienteID INT PRIMARY KEY AUTO_INCREMENT,

    -- Nome do cliente (obrigatório)
    Nome VARCHAR(100) NOT NULL,

    -- Email do cliente (deve ser único)
    Email VARCHAR(100) UNIQUE,

    -- Data em que o cliente foi cadastrado
    DataCadastro DATE,

    -- Restrição adicional (exemplo): idade mínima (se tivéssemos data de nascimento)
    -- CHECK (TIMESTAMPDIFF(YEAR, DataNascimento, CURDATE()) >= 18)
);

-- Inserção de dados de exemplo
INSERT INTO Clientes
    (Nome, Email, DataCadastro)
VALUES
    ('Carlos Pereira', 'carlos.p@mail.net', '2023-11-20'),
    ('Ana Souza', 'ana.souza@provider.org', '2024-02-10');

-- Seleciona clientes cadastrados em 2024
SELECT Nome, Email
FROM Clientes
WHERE YEAR(DataCadastro) = 2024; -- Função YEAR pode variar entre SGBDs
